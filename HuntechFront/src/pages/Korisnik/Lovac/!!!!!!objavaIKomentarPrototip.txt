import React, { useState } from 'react';
import styles from '.NotificationFrame.module.css';

const NotificationFrame = () = {
  const [activeUrl, setActiveUrl] = useState('');

  const notifications = [
    {
      id 'SYS103',
      tekst 'Vaš zahtjev je odobren',
      url 'httpssistem-lovaca.example.comobavjestenjeSYS103',
    },
    {
      id 'SYS104',
      tekst 'Nova aktivnost u sekciji Jaružani',
      url 'httpssistem-lovaca.example.comobavjestenjeSYS104',
    },
    {
      id 'SYS105',
      tekst 'Poziv za učlanjenje u LD Zelengora',
      url 'httpssistem-lovaca.example.comobavjestenjeSYS105',
    },
  ];

  return (
    div className={styles.wrapper}
      h2 className={styles.title}Obavještenjah2

      div className={styles.links}
        {notifications.map((n) = (
          button
            key={n.id}
            className={styles.linkButton}
            onClick={() = setActiveUrl(n.url)}
          
            📄 {n.tekst}
          button
        ))}
      div

      {activeUrl && (
        iframe
          src={activeUrl}
          title=Obavještenje

          height=400px
          className={styles.iframe}

      )}
    div
  );
};

export default NotificationFrame;
